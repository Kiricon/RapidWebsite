    package main

    import rapid "github.com/Kiricon/Rapid"

    func main() {
        // New instance of server
        app := rapid.App()

        // Make a path at "/" that returns "Hello World"
        app.Get("/", func(c rapid.Connection) {
            c.Send("Hello World")
        })

        // Make your server listen on port 8080
        app.Listen(8080)
    }
